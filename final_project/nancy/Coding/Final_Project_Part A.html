#Linear regression for D.C. population and food insecurity 2011-2016


x = np.array([634725, 650431, 662513, 675254, 686575, 695691, 702455])#DC Population
y = np.array([12.1, 12.1, 12.1, 11.1, 11.1, 11.1, 11.1])#Food insecurity percentage in 2011 and 2016


coefficients = np.polyfit(x, y, 1)  # degree 1 = linear
slope, intercept = coefficients

regression_line = slope * x + intercept


plt.scatter(x, y, color='blue', label='Data points')
plt.plot(x, regression_line, color='red', label=f'Linear fit: y = {slope:.2f}x + {intercept:.2f}')
plt.xlabel('DC Population')
plt.ylabel('Food insecurity percentage of population')
plt.legend()
plt.title('Linear Regression between D.C. population and food insecurity 2011-2017')
plt.grid(True)
plt.show()
